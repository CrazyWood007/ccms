# form模板
CMS表单提交模板
常用的提交表单项 <br/>

## 表单提交展示
import Form, { PropsType, PropsTypeFormConfig, PropsTypeStep } from "./";
import CCMS from "../../main";

### 水平展示

<Playground>
  {() => {
    return(
      <div>
         <CCMS
            checkPageAuth={async () => true}
            loadPageURL={async (id) => `/url?id=${id}&type=page`}
            loadPageFrameURL={async (id) => `/url?id=${id}&type=open`}
            loadPageConfig={async (page) => {
                return {
                type: "form",
                layout: "horizontal",
                fields:[]
              }
            }}
            sourceData={{}}
            callback={() => {
              if (window.history.length > 1) {
                window.history.back()
              } else {
                window.close()
              }
            }}
            config={{
              type: "form",
              layout: "horizontal",
              fields:[
              ]
            }}
          />
      </div>
    )
  }}
</Playground>
